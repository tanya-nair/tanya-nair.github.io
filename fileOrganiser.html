            # Scan all the files in particular folder
            # identify the type of the file
            # move the file to another location
            # Repeat this for all the files
            
            import os, glob, shutil
            
            path = '/Users/office/Downloads/'
            
            files = glob.glob(path + '*')
            
            IMAGES_EXT = ['jpg', 'jpeg', 'img', 'gif', 'png']
            DOCS_EXT = ['docx', 'pdf', 'ppt', 'pptx', 'doc']
            MUSIC_EXT = ['mp3', 'wav']
            MOVIE_EXT =['mp4', 'mov', 'mkv']
            MISC_EXT = ['dmg', 'pkg', 'zip', 'rar', 'jar']
            
            
            images = path + 'images'
            docs = path + 'docs'
            music = path + 'music'
            movies = path + 'movies'
            misc = path + 'misc'
            
            for filename in files:
                extension = filename.split('.')[-1]
                if extension.lower() in IMAGES_EXT:
                    shutil.copy2(filename, images)
                if extension.lower() in DOCS_EXT:
                    shutil.copy2(filename, docs)
