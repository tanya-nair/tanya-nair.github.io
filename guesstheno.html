Guess The Number

        
            from random import randint
from os import system

x = randint(1,50)
guesses=5
hasAnswered = False
print("guess the number between 1 and 50, we will indicate whether your guess was more or less. you have 5 chances...use them wisely.")
while guesses > 0:
    user_guess = int(input("Guess the number: "))
    if user_guess > x:
        print("Your guess was more")
        system("say Your guess was more")
    elif user_guess < x:
        print("Your guess was less")
        system("say Your guess was less")
    else:
        print("Your guess was correct")
        system("say Your guess was correct")
        hasAnswered = True
        break
    guesses = guesses - 1

if not hasAnswered:
    output = "You lost, right answer was" + str(x)
    system("say "+ output)
